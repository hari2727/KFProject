import { Test, TestingModule } from '@nestjs/testing';
import { ProfileVersionsService } from './profile-versions.service';
import { ProfileVersionsRepository } from './profile-versions.repository';
import { ProfileVersionsRawResponseMock, ProfileVersionsResponseMock } from './profile-versions.fixtures';
import { HttpException } from '@nestjs/common';

describe('ProfileVersionsService', () => {
    let service: ProfileVersionsService;
    let repository: ProfileVersionsRepository;
    let queryParams: any = {
        successProfileId: 123,
        loggedInUserClientId: 1,
        locale: 'en-US',
        userId: 12334,
    };

    beforeEach(async () => {
        const module: TestingModule = await Test.createTestingModule({
            providers: [ProfileVersionsService, ProfileVersionsRepository],
        }).compile();

        service = module.get<ProfileVersionsService>(ProfileVersionsService);
        repository = module.get<ProfileVersionsRepository>(ProfileVersionsRepository);
    });

    it('should be defined', () => {
        expect(service).toBeDefined();
    });

    it('should call getProfileVersionDetails', async () => {
        jest.spyOn(repository, 'getProfileVersionDetails').mockResolvedValue([]);
        await service.getProfileVersionDetails(queryParams);
        expect(repository.getProfileVersionDetails).toHaveBeenCalledWith(queryParams);
    });

    // it('should throw error in getProfileVersionDetails', async () => {
    //     jest.spyOn(repository, 'getProfileVersionDetails').mockRejectedValue(new Error('Error'));
    //     try {
    //         await service.getProfileVersionDetails(queryParams);
    //     } catch (error) {
    //         expect(error).toBeInstanceOf(HttpException);
    //     }
    // });

    it('should map response data', async () => {
        jest.spyOn(repository, 'getProfileVersionDetails').mockResolvedValueOnce(ProfileVersionsRawResponseMock);
        const response = await service.getProfileVersionDetails(queryParams);
        expect(response).toEqual(ProfileVersionsResponseMock);
    });

    it('should map response data with empty response', async () => {
        jest.spyOn(repository, 'getProfileVersionDetails').mockResolvedValueOnce([]);
        const response = await service.getProfileVersionDetails(queryParams);
        expect(response).toEqual([]);
    });

    it('should throw an error on successProfileId is not a number', async () => {
        try {
            queryParams = {...queryParams, successProfileId: 'abc'};
            await service.getProfileVersionDetails(queryParams);
        } catch (error) {
            expect(error).toBeInstanceOf(HttpException);
        }
    })
});
